Go hỗ trợ các hàm đệ quy. Dưới đây là một ví dụ kinh điển.

 
package main

import "fmt"

Hàm fact này tự gọi chính nó cho đến khi nó rơi vào trường hợp cơ sở là fact(0).

 
func fact(n int) int {
    if n == 0 {
        return 1
    }
    return n * fact(n-1)
}

    fmt.Println(fact(7))
